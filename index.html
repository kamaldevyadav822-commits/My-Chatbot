const chatBox = document.getElementById("chatBox"); const audioEl = document.getElementById("remoteAudio"); async function connect() { const resp = await fetch(SERVER_URL + "/session", { method: "POST" }); const { client_secret } = await resp.json(); const token = client_secret?.value; pc = new RTCPeerConnection(); pc.ontrack = e => audioEl.srcObject = e.streams[0]; dc = pc.createDataChannel("oai-events"); dc.onmessage = e => { const msg = JSON.parse(e.data); if (msg.type === "response.delta") addMsg("assistant", msg.delta); }; pc.addTransceiver("audio", { direction: "recvonly" }); const offer = await pc.createOffer(); await pc.setLocalDescription(offer); const sdp = await fetch("https://api.openai.com/v1/realtime?model=gpt-4o-realtime-preview-2024-12-17", { method: "POST", body: offer.sdp, headers: { Authorization: `Bearer ${token}`, "Content-Type": "application/sdp" } }).then(r => r.text()); await pc.setRemoteDescription({ type: "answer", sdp }); addMsg("system", "Connected to AI"); } function sendText() { const input = document.getElementById("textInput"); const txt = input.value; if (!txt || !dc) return; dc.send(JSON.stringify({ type: "response.create", response: { instructions: txt } })); addMsg("user", txt); input.value = ""; } async function toggleMic() { if (stream) { stream.getTracks().forEach(t => t.stop()); stream = null; return; } stream = await navigator.mediaDevices.getUserMedia({ audio: true }); stream.getTracks().forEach(t => pc.addTrack(t, stream)); } function addMsg(role, text) { const div = document.createElement("div"); div.className = `message ${role}`; div.textContent = `${role}: ${text}`; chatBox.appendChild(div); chatBox.scrollTop = chatBox.scrollHeight; }
